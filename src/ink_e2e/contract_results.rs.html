57// todo use the obj one from `pallet-revive` instead
58/// Result type of a `bare_call`, `bare_instantiate`, `ReviveApi::call`, and
59/// `ReviveApi::instantiate`.
60///
61/// It contains the execution result together with some auxiliary information.
62///
63/// # Note
64///
65/// It has been extended to include `events` at the end of the struct while not bumping
66/// the `ReviveApi` version. Therefore when SCALE decoding a `ContractResult` its
67/// trailing data should be ignored to avoid any potential compatibility issues.
68#[derive(Debug, Clone, Eq, PartialEq, Encode, Decode)]
69pub struct ContractResult<R, Balance> {
70    /// How much weight was consumed during execution.
71    pub gas_consumed: Weight,
72    /// How much weight is required as gas limit in order to execute this call.
73    ///
74    /// This value should be used to determine the weight limit for on-chain execution.
75    ///
76    /// # Note
77    ///
78    /// This can only different from [`Self::gas_consumed`] when weight pre-charging
79    /// is used. Currently, only `seal_call_runtime` makes use of pre-charging.
80    /// Additionally, any `seal_call` or `seal_instantiate` makes use of pre-charging
81    /// when a non-zero `gas_limit` argument is supplied.
82    pub gas_required: Weight,
83    /// How much balance was paid by the origin into the contract's deposit account in
84    /// order to pay for storage.
85    ///
86    /// The storage deposit is never actually charged from the origin in case of
87    /// [`Self::result`] is `Err`. This is because on error all storage changes are
88    /// rolled back including the payment of the deposit.
89    pub storage_deposit: StorageDeposit<Balance>,
90    /// The execution result of the code.
91    pub result: Result<R, DispatchError>,
92}

149impl<E: Environment, EventLog> InstantiationResult<E, EventLog> {
150    /// Returns a call builder for the contract which was instantiated.
151    ///
152    /// # Note
153    ///
154    /// This uses the "default" ABI for calls for the instantiated contract.
155    ///
156    /// The "default" ABI for calls is "ink", unless the ABI is set to "sol"
157    /// in the ink! project's manifest file (i.e. `Cargo.toml`).
158    pub fn call_builder<Contract>(
159        &self,
160    ) -> <Contract as ContractCallBuilder>::Type<ink::env::DefaultAbi>
161    where
162        Contract: ContractCallBuilder,
163        <Contract as ContractCallBuilder>::Type<ink::env::DefaultAbi>: FromAddr,
164    {
165        <<Contract as ContractCallBuilder>::Type<ink::env::DefaultAbi> as FromAddr>::from_addr(self.addr)
166    }
167
168    /// Returns a call builder for the specified ABI for the contract which was
169    /// instantiated.
170    pub fn call_builder_abi<Contract, Abi>(
171        &self,
172    ) -> <Contract as ContractCallBuilder>::Type<Abi>
173    where
174        Contract: ContractCallBuilder,
175        <Contract as ContractCallBuilder>::Type<Abi>: FromAddr,
176    {
177        <<Contract as ContractCallBuilder>::Type<Abi> as FromAddr>::from_addr(self.addr)
178    }
179}

228pub struct CallResult<E: Environment, V, EventLog, Abi = Ink> {
229    /// The result of the dry run, contains debug messages if there were any.
230    pub dry_run: CallDryRunResult<E, V, Abi>,
231    /// Events that happened with the contract instantiation.
232    pub events: EventLog,
233    /// todo

237impl<E: Environment, V: scale::Decode, EventLog> CallResult<E, V, EventLog> {
238    /// Returns the [`MessageResult`] from the execution of the dry-run message
239    /// call.
240    ///
241    /// # Panics
242    /// - if the dry-run message call failed to execute.
243    /// - if message result cannot be decoded into the expected return value type.
244    pub fn message_result(&self) -> MessageResult<V> {
245        self.dry_run.message_result()
246    }
247
248    /// Returns the decoded return value of the message from the dry-run.
249    ///
250    /// Panics if the value could not be decoded. The raw bytes can be accessed
251    /// via [`CallResult::return_data`].
252    pub fn return_value(self) -> V {
253        self.dry_run.return_value()
254    }
255
256    /// Returns the return value of the message dry-run as raw bytes.
257    ///
258    /// Panics if the dry-run message call failed to execute.
259    pub fn return_data(&self) -> &[u8] {
260        &self.dry_run.exec_return_value().data
261    }
262}

306impl<E: Environment, V, Abi> CallDryRunResult<E, V, Abi> {
307    /// Returns true if the dry-run execution resulted in an error.
308    pub fn is_err(&self) -> bool {
309        self.exec_result.result.is_err() || self.did_revert()
310    }
311
312    /// Returns the [`ExecReturnValue`] resulting from the dry-run message call.
313    ///
314    /// Panics if the dry-run message call failed to execute.
315    pub fn exec_return_value(&self) -> &ExecReturnValue {
316        self.exec_result
317            .result
318            .as_ref()
319            .unwrap_or_else(|call_err| panic!("Call dry-run failed: {call_err:?}"))
320    }
321
322    /// Returns true if the message call reverted.
323    pub fn did_revert(&self) -> bool {
324        let res = self.exec_result.result.clone().expect("no result found");
325        res.did_revert()
326    }
327
328    /// Returns the return value as raw bytes of the message from the dry-run.
329    ///
330    /// Panics if the dry-run message call failed to execute.
331    pub fn return_data(&self) -> &[u8] {
332        &self.exec_return_value().data
333    }
334}

336impl<E: Environment, V: scale::Decode> CallDryRunResult<E, V, Ink> {
337    /// Returns the [`MessageResult`] from the execution of the dry-run message call.
338    ///
339    /// # Panics
340    /// - if the dry-run message call failed to execute.
341    /// - if message result cannot be decoded into the expected return value type.
342    pub fn message_result(&self) -> MessageResult<V> {
343        let data = &self.exec_return_value().data;
344        scale::Decode::decode(&mut data.as_ref()).unwrap_or_else(|env_err| {
345            panic!(
346                "Decoding dry run result to ink! message return type failed: {env_err} {:?}",
347                self.exec_return_value()
348            )
349        })
350    }
351
352    /// Returns the decoded return value of the message from the dry-run.
353    ///
354    /// Panics if the value could not be decoded. The raw bytes can be accessed via
355    /// [`CallResult::return_data`].
356    pub fn return_value(&self) -> V {
357        self.message_result()
358            .unwrap_or_else(|lang_err| {
359                panic!(
360                    "Encountered a `LangError` while decoding dry run result to ink! message: {lang_err:?}"
361                )
362            })
363    }
364}

366impl<E: Environment, V: SolDecode> CallDryRunResult<E, V, Sol> {
367    /// Returns the [`MessageResult`] from the execution of the dry-run message call.
368    ///
369    /// # Panics
370    /// - if the dry-run message call failed to execute.
371    /// - if message result cannot be decoded into the expected return value type.
372    pub fn message_result(&self) -> MessageResult<V> {
373        Ok(self.return_value())
374    }
375
376    /// Returns the decoded return value of the message from the dry-run.
377    ///
378    /// Panics if the value could not be decoded. The raw bytes can be accessed via
379    /// [`CallResult::return_data`].
380    pub fn return_value(&self) -> V {
381        // Solidity ABI encoded message calls return data without wrapping it in
382        // `MessageResult`.
383        let data = &self.exec_return_value().data;
384        if self.is_err() {
385            // For Solidity ABI encoded message calls,
386            // the return value is only meaningful if the message call was successful.
387            panic!("Could not decode the dry run result because the message call failed to execute")
388        }
389        SolDecode::decode(data.as_ref()).unwrap_or_else(|err| {
390            panic!(
391                "Encountered an error while decoding dry run result to ink! message: {err:?}"
392            )
393        })
394    }
395}

397/// Result of the dry run of a contract call.
398pub struct InstantiateDryRunResult<E: Environment> {
399    /// The result of the dry run, contains debug messages if there were any.
400    pub contract_result: ContractInstantiateResultFor<E>,
401}
402
403impl<E: Environment> From<ContractInstantiateResultFor<E>>
404    for InstantiateDryRunResult<E>
405{
406    fn from(contract_result: ContractInstantiateResultFor<E>) -> Self {
407        Self { contract_result }
408    }
409}
410
411impl<E: Environment> InstantiateDryRunResult<E> {
412    /// Returns true if the dry-run execution resulted in an error.
413    pub fn is_err(&self) -> bool {
414        self.contract_result.result.is_err()
415    }
416
417    /// Returns the [`InstantiateReturnValue`] resulting from the dry-run message call.
418    ///
419    /// Panics if the dry-run message call failed to execute.
420    pub fn instantiate_return_value(&self) -> &InstantiateReturnValue {
421        self.contract_result
422            .result
423            .as_ref()
424            .unwrap_or_else(|call_err| panic!("Instantiate dry-run failed: {call_err:?}"))
425    }
426
427    /// Returns the encoded return value from the constructor.
428    ///
429    /// # Panics
430    /// - if the dry-run message instantiate failed to execute.
431    /// - if message result cannot be decoded into the expected return value type.
432    pub fn constructor_result<V: scale::Decode>(&self) -> ConstructorResult<V> {
433        let data = &self.instantiate_return_value().result.data;
434        scale::Decode::decode(&mut data.as_ref()).unwrap_or_else(|env_err| {
435            panic!("Decoding dry run result to constructor return type failed: {env_err}")
436        })
437    }
438
439    /// Returns the return value of the instantiation dry-run as raw bytes.
440    ///
441    /// Panics if the dry-run message call failed to execute.
442    pub fn return_data(&self) -> &[u8] {
443        &self.instantiate_return_value().result.data
444    }
445
446    /// todo
447    pub fn did_revert(&self) -> bool {
448        let res = self.instantiate_return_value().clone().result;
449        res.did_revert()
450    }
451}
